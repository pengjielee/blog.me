## 实现1:

```
const reverse1 = (str) => {
  return str.split("").reverse().join("");
};
```

## 实现2：

```
const reverse2 = (str) => {
  let result = "";
  for (let char of str) {
    result = char + result;
  }
  return result;
};
```

## 实现3：

```
const reverse3 = (str) => {
  return str.split("").reduce((result, char) => char + result);
};
```
