如何判断左右小括号是否全部匹配。如 (( ))()((((()))))？

```
var isValid = function (s) {
  var stack = [];
  var map = {
    "(": ")",
    "[": "]",
    "{": "}",
  };

  for (var char of s) {
    if (char in map) {
      stack.push(char);
    } else {
      if (!stack.length || char != map[stack.pop()]) {
        return false;
      }
    }
  }

  // 如果最后stack 里没有元素了， 就一定是匹配的
  return !stack.length;
};
```
